#include <stdio.h>

int main() {
    char nomes[7][50];
    float medias[7];
    int i;
    float maior = 0;
    int idxMaior = 0;

    for(i = 0; i < 7; i++) {
        printf("Digite o nome do aluno %d: ", i+1);
        scanf(" %[^\n]", nomes[i]);
        printf("Digite a média final: ");
        scanf("%f", &medias[i]);

        if(medias[i] > maior) {
            maior = medias[i];
            idxMaior = i;
        }
    }

    printf("Aluno com maior média: %s\n", nomes[idxMaior]);

    for(i = 0; i < 7; i++) {
        if(medias[i] < 7) {
            float notaExame = 5*2 - medias[i]; // Média exame 5
            printf("%s precisa tirar %.2f no exame final\n", nomes[i], notaExame);
        }
    }

    return 0;
}
