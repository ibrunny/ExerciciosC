#include <stdio.h>

int main() {
    int vetor[7], i;

    printf("Digite 7 números inteiros:\n");
    for(i = 0; i < 7; i++)
        scanf("%d", &vetor[i]);

    printf("Múltiplos de 2: ");
    for(i = 0; i < 7; i++)
        if(vetor[i] % 2 == 0) printf("%d ", vetor[i]);

    printf("\nMúltiplos de 3: ");
    for(i = 0; i < 7; i++)
        if(vetor[i] % 3 == 0) printf("%d ", vetor[i]);

    printf("\nMúltiplos de 2 e 3: ");
    for(i = 0; i < 7; i++)
        if(vetor[i] % 2 == 0 && vetor[i] % 3 == 0) printf("%d ", vetor[i]);

    printf("\n");
    return 0;
}
