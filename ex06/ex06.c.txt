#include <stdio.h>

int main() {
    char nomes[10][50];
    float vendas[10], comissao[10], totalVendas = 0, maior, menor;
    int i, idxMaior = 0, idxMenor = 0;

    for(i = 0; i < 10; i++) {
        printf("Digite o nome do vendedor %d: ", i+1);
        scanf(" %[^\n]", nomes[i]);
        printf("Digite o total de vendas: ");
        scanf("%f", &vendas[i]);
        printf("Digite o percentual de comissão: ");
        scanf("%f", &comissao[i]);
    }

    maior = comissao[0] * vendas[0] / 100;
    menor = comissao[0] * vendas[0] / 100;

    printf("\nRelatório de comissões:\n");
    for(i = 0; i < 10; i++) {
        float valor = vendas[i] * comissao[i] / 100;
        printf("%s: %.2f\n", nomes[i], valor);
        totalVendas += vendas[i];
        if(valor > maior) { maior = valor; idxMaior = i; }
        if(valor < menor) { menor = valor; idxMenor = i; }
    }

    printf("\nTotal das vendas: %.2f\n", totalVendas);
    printf("Maior comissão: %.2f (%s)\n", maior, nomes[idxMaior]);
    printf("Menor comissão: %.2f (%s)\n", menor, nomes[idxMenor]);

    return 0;
}
