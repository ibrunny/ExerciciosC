#include <stdio.h>

int main() {
    int M[4][6], N[6][4], i, j, k, resultado[4][4] = {0};

    printf("Digite a matriz M (4x6):\n");
    for(i=0;i<4;i++)
        for(j=0;j<6;j++)
            scanf("%d",&M[i][j]);

    printf("Digite a matriz N (6x4):\n");
    for(i=0;i<6;i++)
        for(j=0;j<4;j++)
            scanf("%d",&N[i][j]);

    for(i=0;i<4;i++){
        for(j=0;j<4;j++){
            for(k=0;k<6;k++){
                resultado[i][j] += M[i][k] + N[k][j];
            }
        }
    }

    printf("\nResultado:\n");
    for(i=0;i<4;i++){
        for(j=0;j<4;j++){
            printf("%d ", resultado[i][j]);
        }
        printf("\n");
    }

    return 0;
}
