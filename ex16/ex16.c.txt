#include <stdio.h>

int main() {
    int mat[20][10], somaCol[10] = {0}, i, j;

    printf("Digite os elementos da matriz 20x10:\n");
    for(i = 0; i < 20; i++){
        for(j = 0; j < 10; j++){
            scanf("%d", &mat[i][j]);
            somaCol[j] += mat[i][j];
        }
    }

    printf("\nMatriz resultante (cada elemento multiplicado pela soma da coluna):\n");
    for(i = 0; i < 20; i++){
        for(j = 0; j < 10; j++){
            printf("%d ", mat[i][j] * somaCol[j]);
        }
        printf("\n");
    }

    return 0;
}
